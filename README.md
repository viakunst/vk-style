# VK-Style

> Common React components for ViaKunst

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'vk-style'
import 'vk-style/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
